          return false
        }
      })
      function clearPop(){
        $('.pop-img').addClass('pop-hide');
        $('.pop-img').removeClass('pop-show')
        setTimeout(function(){
          $('.pop-img img').attr('src','')
          $('.medium-zoom-image').removeClass('pop-img-click')
          popImg = false
        },300)
      }
      window.onscroll = function(){
        if(popImg){
          clearPop()
        }else{
          return false
        }
      }
    });
